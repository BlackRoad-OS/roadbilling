Easy Integration

// Install the SDK npm install @blackroad/roadbilling // Initialize billing in your app import { RoadBilling } from '@blackroad/roadbilling'; const billing = new RoadBilling({ apiKey: 'sk_live_your_key_here' }); // Track usage automatically await billing.trackUsage({ product: 'roadapi', metric: 'requests', quantity: 1 }); // Get current spending const spending = await billing.getCurrentSpend(); console.log(`Current month: $${spending.amount}`); // Create a checkout session const session = await billing.createCheckout({ products: ['roadauth-pro', 'roadapi-pro'], successUrl: 'https://app.blackroad.io/success' }); // Redirect to Stripe Checkout window.location.href = session.url;
